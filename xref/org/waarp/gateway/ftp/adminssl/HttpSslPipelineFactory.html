24  import org.jboss.netty.channel.Channels;

32  import org.waarp.common.crypto.ssl.WaarpSecureKeyStore;
33  import org.waarp.common.crypto.ssl.WaarpSslContextFactory;
34  import org.waarp.gateway.ftp.config.FileBasedConfiguration;

37   * @author Frederic Bregier

39   */
40  public class HttpSslPipelineFactory implements ChannelPipelineFactory {
41  	public static WaarpSslContextFactory waarpSslContextFactory;
42  	public static WaarpSecureKeyStore WaarpSecureKeyStore;
43  	private final ExecutorService executorService;
44  	public boolean useHttpCompression = false;
45  	public boolean enableRenegotiation = false;
46  
47  	public HttpSslPipelineFactory(boolean useHttpCompression,
48  			boolean enableRenegotiation,
49  			ExecutorService executor) {
50  		this.useHttpCompression = useHttpCompression;
51  		this.enableRenegotiation = enableRenegotiation;
52  		this.executorService = executor;
53  	}
54  
55  	@Override
56  	public ChannelPipeline getPipeline() {
57  		final ChannelPipeline pipeline = Channels.pipeline();
58  		// Add SSL handler first to encrypt and decrypt everything.
59          SslHandler sslhandler = waarpSslContextFactory.initPipelineFactory(true,
60  				false,
61  				enableRenegotiation, executorService);
62          sslhandler.setIssueHandshake(true);
63          pipeline.addLast("ssl", sslhandler);
64  
65  		pipeline.addLast("decoder", new HttpRequestDecoder());
66  		pipeline.addLast("aggregator", new HttpChunkAggregator(1048576));
67  		pipeline.addLast("encoder", new HttpResponseEncoder());
68  		pipeline.addLast("pipelineExecutor", new ExecutionHandler(
69  				FileBasedConfiguration.fileBasedConfiguration.getHttpPipelineExecutor()));
70          pipeline.addLast("streamer", new ChunkedWriteHandler());
71  		if (useHttpCompression) {
72  			pipeline.addLast("deflater", new HttpContentCompressor());
73  		}
74  		pipeline.addLast("handler", new HttpSslHandler());
75  		return pipeline;
76  	}
77  }
