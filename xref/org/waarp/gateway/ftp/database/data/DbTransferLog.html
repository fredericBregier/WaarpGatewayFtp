346 	/**
347 	 * Create a Special Id for NoDb client
348 	 */
349 	private void createNoDbSpecialId() {
350 		synchronized (clientNoDbSpecialId) {
351 			// New SpecialId is not possible with No Database Model
352 			specialId = System.currentTimeMillis();
353 			Long newOne = specialId;
354 			while (clientNoDbSpecialId.contains(newOne)) {
355 				newOne = specialId++;
356 			}
357 			clientNoDbSpecialId.add(newOne);
358 		}
359 	}
360 
361 	/**
362 	 * Remove a Special Id for NoDb Client
363 	 */
364 	private void removeNoDbSpecialId() {
365 		synchronized (clientNoDbSpecialId) {
366 			Long oldOne = specialId;
367 			clientNoDbSpecialId.remove(oldOne);
368 		}
369 	}

385 	 * (non-Javadoc)
386 	 * @see org.waarp.openr66.databaseold.data.AbstractDbData#insert()
387 	 */
388 	@Override
389 	public void insert() throws WaarpDatabaseException {
390 		if (isSaved) {
391 			return;
392 		}
393 		if (dbSession == null) {
394 			if (specialId == DbConstant.ILLEGALVALUE) {
395 				// New SpecialId is not possible with No Database Model
396 				createNoDbSpecialId();
397 			}
398 			isSaved = true;
399 			return;
400 		}
401 		// First need to find a new id if id is not ok
402 		if (specialId == DbConstant.ILLEGALVALUE) {
403 			specialId = DbModelFactory.dbModel.nextSequence(dbSession);
404 			logger.debug("Try Insert create a new Id from sequence: " +
405 					specialId);
406 			setPrimaryKey();
407 		}
408 		super.insert();
409 	}
410 
411 	/**
412 	 * As insert but with the ability to change the SpecialId
413 	 * 
414 	 * @throws WaarpDatabaseException
415 	 */
416 	public void create() throws WaarpDatabaseException {
417 		if (isSaved) {
418 			return;
419 		}
420 		if (dbSession == null) {
421 			if (specialId == DbConstant.ILLEGALVALUE) {
422 				// New SpecialId is not possible with No Database Model
423 				createNoDbSpecialId();
424 			}
425 			isSaved = true;
426 			return;
427 		}
428 		// First need to find a new id if id is not ok
429 		if (specialId == DbConstant.ILLEGALVALUE) {
430 			specialId = DbModelFactory.dbModel.nextSequence(dbSession);
431 			logger.debug("Try Insert create a new Id from sequence: " +
432 					specialId);
433 			setPrimaryKey();
434 		}
435 		setToArray();
436 		DbPreparedStatement preparedStatement = new DbPreparedStatement(
437 				dbSession);
438 		try {
439 			preparedStatement.createPrepareStatement("INSERT INTO " + table +
440 					" (" + selectAllFields + ") VALUES " + insertAllValues);
441 			setValues(preparedStatement, allFields);
442 			try {
443 				int count = preparedStatement.executeUpdate();
444 				if (count <= 0) {
445 					throw new WaarpDatabaseNoDataException("No row found");
446 				}
447 			} catch (WaarpDatabaseSqlException e) {
448 				logger.error("Problem while inserting", e);
449 				DbPreparedStatement find = new DbPreparedStatement(dbSession);
450 				try {
451 					find.createPrepareStatement("SELECT MAX(" +
452 							primaryKey[3].column + ") FROM " + table + " WHERE " +
453 							primaryKey[0].column + " = ? AND " +
454 							primaryKey[1].column + " = ? AND " +
455 							primaryKey[2].column + " = ? AND " +
456 							primaryKey[3].column + " != ? ");
457 					setPrimaryKey();
458 					setValues(find, primaryKey);
459 					find.executeQuery();
460 					if (find.getNext()) {
461 						long result;
462 						try {
463 							result = find.getResultSet().getLong(1);
464 						} catch (SQLException e1) {
465 							throw new WaarpDatabaseSqlException(e1);
466 						}
467 						specialId = result + 1;
468 						DbModelFactory.dbModel.resetSequence(dbSession, specialId + 1);
469 						setToArray();
470 						preparedStatement.close();
471 						setValues(preparedStatement, allFields);
472 						int count = preparedStatement.executeUpdate();
473 						if (count <= 0) {
474 							throw new WaarpDatabaseNoDataException("No row found");
475 						}
476 					} else {
477 						throw new WaarpDatabaseNoDataException("No row found");
478 					}
479 				} finally {
480 					find.realClose();
481 				}
482 			}
483 			isSaved = true;
484 		} finally {
485 			preparedStatement.realClose();
486 		}
487 	}

523 	 * @throws WaarpDatabaseNoConnectionException

526 	public static DbPreparedStatement getStatusPrepareStament(
527 			DbSession session, ReplyCode status, int limit)
528 			throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
529 		String request = "SELECT " + selectAllFields + " FROM " + table;
530 		if (status != null) {
531 			request += " WHERE " + Columns.INFOSTATUS.name() + " = " +
532 					status.getCode() + " AND " + getLimitWhereCondition();
533 		} else {
534 			request += " WHERE " + getLimitWhereCondition();
535 		}
536 		request += " ORDER BY " + Columns.STARTTRANS.name() + " DESC ";
537 		if (limit > 0) {
538 			request = DbModelFactory.dbModel.limitRequest(selectAllFields, request, limit);
539 		}
540 		return new DbPreparedStatement(session, request);
541 	}

561 			preparedStatement.createPrepareStatement(request);
562 			try {
563 				preparedStatement.getPreparedStatement().setTimestamp(1, start);
564 				preparedStatement.getPreparedStatement().setTimestamp(2, stop);
565 			} catch (SQLException e) {
566 				preparedStatement.realClose();

570 			request += " WHERE " + Columns.STARTTRANS.name() +
571 					" >= ? AND " + getLimitWhereCondition() +
572 					" ORDER BY " + Columns.SPECIALID.name() + " DESC ";
573 			preparedStatement.createPrepareStatement(request);

575 				preparedStatement.getPreparedStatement().setTimestamp(1, start);
576 			} catch (SQLException e) {
577 				preparedStatement.realClose();
578 				throw new WaarpDatabaseSqlException(e);
579 			}
580 		} else if (stop != null) {
581 			request += " WHERE " + Columns.STARTTRANS.name() +
582 					" <= ? AND " + getLimitWhereCondition() +
583 					" ORDER BY " + Columns.SPECIALID.name() + " DESC ";
584 			preparedStatement.createPrepareStatement(request);

593 					" ORDER BY " + Columns.SPECIALID.name() + " DESC ";

624 	public static long getResultCountPrepareStatement(DbPreparedStatement pstt, UpdatedInfo info,
625 			long time) {
626 		long result = 0;
627 		try {
628 			finishSelectOrCountPrepareStatement(pstt, time);
629 			pstt.getPreparedStatement().setInt(2, info.ordinal());
630 			pstt.executeQuery();
631 			if (pstt.getNext()) {
632 				result = pstt.getResultSet().getLong(1);
633 			}
634 		} catch (WaarpDatabaseNoConnectionException e) {
635 		} catch (WaarpDatabaseSqlException e) {
636 		} catch (SQLException e) {
637 		} finally {
638 			pstt.close();
