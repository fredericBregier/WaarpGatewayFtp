202 	/**
203 	 * Unregister and Register again the SNMPv2MIB with System adapted to this Mib
204 	 * 
205 	 * @throws DuplicateRegistrationException
206 	 */
207 	protected void agentRegisterSystem() throws DuplicateRegistrationException {
208 		// Since BaseAgent registers some mibs by default we need to unregister
209 		// one before we register our own sysDescr. Normally you would
210 		// override that method and register the mibs that you need
211 
212 		agent.unregisterManagedObject(agent.getSnmpv2MIB());
213 
214 		// Register a system description, use one from you product environment
215 		// to test with
216 		snmpv2 = new SNMPv2MIB(new OctetString(SnmpName), ggObjectId,
217 				new Integer32(SnmpService));
218 		snmpv2.setContact(new OctetString(SnmpDefaultAuthor));
219 		snmpv2.setLocation(new OctetString(SnmpDefaultLocalization));
220 		snmpv2.setName(new OctetString(SnmpVersion));
221 		snmpv2.registerMOs(agent.getServer(), null);
222 		if (logger.isDebugEnabled()) {
223 			logger.debug("SNMPV2: " + snmpv2.getContact() + ":" +
224 					snmpv2.getDescr() + ":" + snmpv2.getLocation() + ":" +
225 					snmpv2.getName() + ":" + snmpv2.getObjectID() + ":" +
226 					snmpv2.getServices() + ":" + snmpv2.getUpTime());
227 		}
228 		// Save UpTime reference since used everywhere
229 		upTime = snmpv2.getSysUpTime();
230 	}

232 	/**
233 	 * Register this MIB
234 	 * 
235 	 * @throws DuplicateRegistrationException
236 	 */
237 	protected void agentRegisterWaarpMib()
238 			throws DuplicateRegistrationException {
239 		logger.debug("registerGGMib");
240 		// register Static info
241 		rowInfo = new WaarpMORow(this, rootOIDWaarpInfo,
242 				WaarpDefinition, MibLevel.staticInfo.ordinal());
243 		rowInfo.setValue(WaarpDefinitionIndex.applName.ordinal(),
244 				"Waarp OpenR66");
245 		rowInfo.setValue(WaarpDefinitionIndex.applServerName.ordinal(),
246 				FileBasedConfiguration.fileBasedConfiguration.HOST_ID);
247 		rowInfo.setValue(WaarpDefinitionIndex.applVersion.ordinal(),
248 				Version.ID);
249 		rowInfo.setValue(WaarpDefinitionIndex.applDescription.ordinal(),
250 				"Waarp OpenR66: File Transfer Monitor");
251 		rowInfo.setValue(WaarpDefinitionIndex.applURL.ordinal(),
252 				"http://waarp.github.com/Waarp");
253 		rowInfo.setValue(
254 				WaarpDefinitionIndex.applApplicationProtocol.ordinal(),
255 				applicationProtocol);
256 
257 		rowInfo.registerMOs(agent.getServer(), null);
258 		// register General info
259 		rowGlobal = new WaarpMORow(this, rootOIDWaarpGlobal,
260 				WaarpGlobalValues, MibLevel.globalInfo.ordinal());
261 		WaarpMOScalar memoryScalar = rowGlobal.row[WaarpGlobalValuesIndex.memoryTotal
262 				.ordinal()];
263 		memoryScalar.setValue(new MemoryGauge32(MemoryType.TotalMemory));
264 		memoryScalar = rowGlobal.row[WaarpGlobalValuesIndex.memoryFree
265 				.ordinal()];
266 		memoryScalar.setValue(new MemoryGauge32(MemoryType.FreeMemory));
267 		memoryScalar = rowGlobal.row[WaarpGlobalValuesIndex.memoryUsed
268 				.ordinal()];
269 		memoryScalar.setValue(new MemoryGauge32(MemoryType.UsedMemory));
270 		rowGlobal.registerMOs(agent.getServer(), null);
271 		// setup UpTime to SysUpTime and change status
272 		scalarUptime = rowGlobal.row[WaarpGlobalValuesIndex.applUptime
273 				.ordinal()];
274 		scalarUptime.setValue(new WaarpUptime(upTime));
275 		changeStatus(OperStatus.restarting);
276 		changeStatus(OperStatus.up);
277 		// register Detailed info
278 		rowDetailed = new WaarpMORow(this, rootOIDWaarpDetailed,
279 				WaarpDetailedValues, MibLevel.detailedInfo.ordinal());
280 		rowDetailed.registerMOs(agent.getServer(), null);
281 		// register Error info
282 		rowError = new WaarpMORow(this, rootOIDWaarpError,
283 				WaarpErrorValues, MibLevel.errorInfo.ordinal());
284 		rowError.registerMOs(agent.getServer(), null);
285 	}

364 	/**
365 	 * Change the status and the LastChange Timeticks
366 	 * 
367 	 * @param status
368 	 */
369 	public void changeStatus(OperStatus status) {
370 		WaarpMOScalar statusScalar = rowGlobal.row[WaarpGlobalValuesIndex.applOperStatus
371 				.ordinal()];
372 		Integer32 var = (Integer32) statusScalar.getValue();
373 		if (var.getValue() != status.status) {
374 			var.setValue(status.status);
375 			WaarpMOScalar lastTimeScalar = rowGlobal.row[WaarpGlobalValuesIndex.applLastChange
376 					.ordinal()];
377 			TimeTicks time = (TimeTicks) lastTimeScalar.getValue();
378 			time.setValue(upTime.get().getValue());
379 		}
380 	}

430 	/**
431 	 * Send a notification (trap or inform) for Warning/Error event on a single Transfer Task
432 	 * 
433 	 * @param message
434 	 * @param runner
435 	 */
436 	public void notifyInfoTask(String message, DbTransferLog runner) {
437 		if (!TrapLevel.All.isLevelValid(agent.trapLevel))
438 			return;
439 		if (logger.isDebugEnabled())
440 			logger.debug("Notify: " + NotificationElements.InfoTask + ":" +
441 					message + ":" + runner);
442 		long delay = (runner.getStart().getTime() - agent.getUptimeSystemTime()) / 10;
443 		if (delay < 0)
444 			delay = 0;
445 		agent.getNotificationOriginator().notify(
446 				new OctetString("public"),
447 				NotificationElements.InfoTask.getOID(rootOIDWaarpNotif),
448 				new VariableBinding[] {
449 						new VariableBinding(NotificationElements.InfoTask
450 								.getOID(rootOIDWaarpNotif, 1),
451 								new OctetString(NotificationElements.InfoTask
452 										.name())),
453 						new VariableBinding(NotificationElements.InfoTask
454 								.getOID(rootOIDWaarpNotif, 1),
455 								new OctetString(message)),
456 						// Start of Task
457 						new VariableBinding(
458 								NotificationElements.InfoTask
459 										.getOID(rootOIDWaarpNotif,
460 												NotificationTasks.filenameInfo
461 														.getOID()),
462 								new OctetString(runner.getFilename())),
463 						new VariableBinding(NotificationElements.InfoTask
464 								.getOID(rootOIDWaarpNotif,
465 										NotificationTasks.modeTransInfo
466 												.getOID()),
467 								new OctetString(runner.getMode())),
468 						new VariableBinding(NotificationElements.InfoTask
469 								.getOID(rootOIDWaarpNotif,
470 										NotificationTasks.startTransInfo
471 												.getOID()),
472 								new TimeTicks(delay)),
473 						new VariableBinding(NotificationElements.InfoTask
474 								.getOID(rootOIDWaarpNotif,
475 										NotificationTasks.infoStatusInfo
476 												.getOID()), new OctetString(
477 								runner.getErrorInfo().getMesg())),
478 						new VariableBinding(NotificationElements.InfoTask
479 								.getOID(rootOIDWaarpNotif,
480 										NotificationTasks.userIdInfo
481 												.getOID()), new OctetString(
482 								runner.getUser())),
483 						new VariableBinding(NotificationElements.InfoTask
484 								.getOID(rootOIDWaarpNotif,
485 										NotificationTasks.accountId
486 												.getOID()), new OctetString(
487 								runner.getAccount())),
488 						new VariableBinding(NotificationElements.InfoTask
489 								.getOID(rootOIDWaarpNotif,
490 										NotificationTasks.specialIdInfo
491 												.getOID()), new OctetString("" +
492 								runner.getSpecialId())),
493 						// End of Task
494 						new VariableBinding(SnmpConstants.sysDescr, snmpv2
495 								.getDescr()),
496 						new VariableBinding(SnmpConstants.sysObjectID, snmpv2
497 								.getObjectID()),
498 						new VariableBinding(SnmpConstants.sysContact, snmpv2
499 								.getContact()),
500 						new VariableBinding(SnmpConstants.sysName, snmpv2
501 								.getName()),
502 						new VariableBinding(SnmpConstants.sysLocation, snmpv2
503 								.getLocation()) });
504 	}

507 	 * Trap/Notification
508 	 * 
509 	 * @param element
510 	 * @param message
511 	 * @param message2
512 	 */
513 	private void notify(NotificationElements element, String message,
514 			String message2) {
515 		if (logger.isDebugEnabled())
516 			logger.debug("Notify: " + element + ":" + message + ":" + message2);
517 		agent.getNotificationOriginator().notify(
518 				new OctetString("public"),
519 				element.getOID(rootOIDWaarpNotif),
520 				new VariableBinding[] {
521 						new VariableBinding(element.getOID(
522 								rootOIDWaarpNotif, 1), new OctetString(
523 								element.name())),
524 						new VariableBinding(element.getOID(
525 								rootOIDWaarpNotif, 1), new OctetString(
526 								message)),
527 						new VariableBinding(element.getOID(
528 								rootOIDWaarpNotif, 1), new OctetString(
529 								message2)),
530 						new VariableBinding(SnmpConstants.sysDescr, snmpv2
531 								.getDescr()),
532 						new VariableBinding(SnmpConstants.sysObjectID, snmpv2
533 								.getObjectID()),
534 						new VariableBinding(SnmpConstants.sysContact, snmpv2
535 								.getContact()),
536 						new VariableBinding(SnmpConstants.sysName, snmpv2
537 								.getName()),
538 						new VariableBinding(SnmpConstants.sysLocation, snmpv2
539 								.getLocation()) });
540 	}
