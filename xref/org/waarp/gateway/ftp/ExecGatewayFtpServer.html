31  /**
32   * Exec FTP Server using simple authentication (XML FileInterface based), and standard Directory and
33   * FileInterface implementation (Filesystem based).
34   * 
35   * @author Frederic Bregier
36   * 
37   */
38  public class ExecGatewayFtpServer {
39  	/**
40  	 * Internal Logger
41  	 */
42  	private static WaarpInternalLogger logger = null;
43  
44  	/**
45  	 * Take a simple XML file as configuration.
46  	 * 
47  	 * @param args
48  	 */
49  	public static void main(String[] args) {
50  		if (args.length < 1) {
51  			System.err.println("Usage: " +
52  					ExecGatewayFtpServer.class.getName() + " <config-file> [<r66config-file>]");
53  			return;
54  		}
55  		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
56  		logger = WaarpInternalLoggerFactory
57  				.getLogger(ExecGatewayFtpServer.class);
58  		String config = args[0];
59  		FileBasedConfiguration configuration = new FileBasedConfiguration(
60  				ExecGatewayFtpServer.class, ExecBusinessHandler.class,
61  				FileSystemBasedDataBusinessHandler.class,
62  				new FilesystemBasedFileParameterImpl());
63  		if (!configuration.setConfigurationServerFromXml(config)) {
64  			System.err.println("Bad main configuration");
65  			return;
66  		}
67  		Configuration.configuration.useLocalExec = configuration.useLocalExec;
68  		if (AbstractExecutor.useDatabase) {
69  			// Use R66 module
70  			if (args.length > 1) {
71  				if (!org.waarp.openr66.configuration.FileBasedConfiguration
72  						.setSubmitClientConfigurationFromXml(Configuration.configuration,
73  								args[1])) {
74  					// if (!R66FileBasedConfiguration.setSimpleClientConfigurationFromXml(args[1]))
75  					// {
76  					System.err.println("Bad R66 configuration");
77  					return;
78  				}
79  			} else {
80  				// Cannot get R66 functional
81  				System.err.println("No R66PrepareTransfer configuration file");
82  			}
83  		}
84  		FileBasedConfiguration.fileBasedConfiguration = configuration;
85  		// Start server.
86  		configuration.configureLExec();
87  		configuration.serverStartup();
88  		configuration.configureHttps();
89  		configuration.configureConstraint();
90  		try {
91  			configuration.configureSnmp();
92  		} catch (FtpNoConnectionException e) {
93  			System.err.println("Cannot start SNMP support: " + e.getMessage());
94  		}
95  		logger.warn("FTP started "+ 
96  				(configuration.getFtpInternalConfiguration().isUsingNativeSsl()? "Implicit SSL On" :
97  					configuration.getFtpInternalConfiguration().isAcceptAuthProt() ?"Explicit SSL On":
98  						"with SSL Off"));
99  	}
100 
101 }
