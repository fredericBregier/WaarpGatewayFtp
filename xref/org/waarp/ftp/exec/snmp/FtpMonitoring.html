195 	private static final int ref421 =
196 			ReplyCode.REPLY_421_SERVICE_NOT_AVAILABLE_CLOSING_CONTROL_CONNECTION.ordinal();

198 	/**
199 	 * Update the reply code counter for other operations than a transfer
200 	 * 
201 	 * @param code
202 	 */
203 	public void updateCodeNoTransfer(ReplyCode code) {
204 		int i = code.ordinal();
205 		if (i >= ref421) {
206 			i -= ref421;
207 			reply_error_notransfers[i]++;
208 		} else {
209 			reply_info_notransfers[i]++;
210 		}
211 	}

227 	/**
228 	 * Update the value for one particular MIB entry
229 	 * 
230 	 * @param type
231 	 * @param entry
232 	 */
233 	public void run(int type, int entry) {
234 		long nbMs = FileBasedConfiguration.fileBasedConfiguration.agentSnmp.getUptime() + 100;
235 		MibLevel level = MibLevel.values()[type];
236 		switch (level) {
237 			case globalInfo:// Global
238 				if (((FtpPrivateMib) this.agent.mib).rowGlobal != null)
239 					run(nbMs, WaarpGlobalValuesIndex.values()[entry]);
240 				return;
241 			case detailedInfo:// Detailed
242 				if (((FtpPrivateMib) this.agent.mib).rowDetailed != null)
243 					run(nbMs, WaarpDetailedValuesIndex.values()[entry]);
244 				return;
245 			case errorInfo:// Error
246 				if (((FtpPrivateMib) this.agent.mib).rowError != null)
247 					run(nbMs, WaarpErrorValuesIndex.values()[entry]);
248 				return;
249 			case staticInfo:
250 				break;
251 			case trapInfo:
252 				break;
253 			default:
254 				break;
255 		}
256 	}

288 	/**
289 	 * Update a value in Global MIB part
290 	 * 
291 	 * @param nbMs
292 	 * @param entry
293 	 */
294 	protected void run(long nbMs, WaarpGlobalValuesIndex entry) {
295 		synchronized (trafficCounter) {
296 			long val = 0;
297 			long limitDate = System.currentTimeMillis() - nbMs;
298 			// Global
299 			try {
300 				switch (entry) {
301 					case applUptime:
302 						return;
303 					case applOperStatus:
304 						return;
305 					case applLastChange:
306 						return;
307 					case applInboundAssociations:
308 						DbTransferLog.finishSelectOrCountPrepareStatement(countInActiveTransfer,
309 								limitDate);
310 						nbInActiveTransfer = DbTransferLog
311 								.getResultCountPrepareStatement(countInActiveTransfer);
312 						updateGlobalValue(entry.ordinal(), nbInActiveTransfer);
313 						return;
314 					case applOutboundAssociations:
315 						DbTransferLog.finishSelectOrCountPrepareStatement(countOutActiveTransfer,
316 								limitDate);
317 						nbOutActiveTransfer = DbTransferLog
318 								.getResultCountPrepareStatement(countOutActiveTransfer);
319 						updateGlobalValue(entry.ordinal(), nbOutActiveTransfer);
320 						return;
321 					case applAccumInboundAssociations:
322 						DbTransferLog.finishSelectOrCountPrepareStatement(countInTotalTransfer,
323 								limitDate);
324 						nbInTotalTransfer = DbTransferLog
325 								.getResultCountPrepareStatement(countInTotalTransfer);
326 						updateGlobalValue(entry.ordinal(), nbInTotalTransfer);
327 						return;
328 					case applAccumOutboundAssociations:
329 						DbTransferLog.finishSelectOrCountPrepareStatement(countOutTotalTransfer,
330 								limitDate);
331 						nbOutTotalTransfer = DbTransferLog
332 								.getResultCountPrepareStatement(countOutTotalTransfer);
333 						updateGlobalValue(entry.ordinal(), nbOutTotalTransfer);
334 						return;
335 					case applLastInboundActivity:
336 						val = (lastInActiveTransfer -
337 								this.agent.getUptimeSystemTime()) / 10;
338 						if (val < 0)
339 							val = 0;
340 						updateGlobalValue(entry.ordinal(), val);
341 						return;
342 					case applLastOutboundActivity:
343 						val = (lastOutActiveTransfer -
344 								this.agent.getUptimeSystemTime()) / 10;
345 						if (val < 0)
346 							val = 0;
347 						updateGlobalValue(entry.ordinal(), val);
348 						return;
349 					case applRejectedInboundAssociations:
350 						DbTransferLog.finishSelectOrCountPrepareStatement(countInErrorTransfer,
351 								limitDate);
352 						nbInErrorTransfer = DbTransferLog
353 								.getResultCountPrepareStatement(countInErrorTransfer);
354 						updateGlobalValue(entry.ordinal(), nbInErrorTransfer);
355 						return;
356 					case applFailedOutboundAssociations:
357 						DbTransferLog.finishSelectOrCountPrepareStatement(countOutErrorTransfer,
358 								limitDate);
359 						nbOutErrorTransfer = DbTransferLog
360 								.getResultCountPrepareStatement(countOutErrorTransfer);
361 						updateGlobalValue(entry.ordinal(), nbOutErrorTransfer);
362 						return;
363 					case applInboundBandwidthKBS:
364 						val = trafficCounter.getLastReadThroughput() >> 10;// B/s -> KB/s
365 						updateGlobalValue(entry.ordinal(), val);
366 						return;
367 					case applOutboundBandwidthKBS:
368 						val = trafficCounter.getLastWriteThroughput() >> 10;
369 						updateGlobalValue(entry.ordinal(), val);
370 						return;
371 					case nbInfoUnknown:
372 						nbCountInfoUnknown = DbTransferLog.getResultCountPrepareStatement(
373 								countInfo,
374 								UpdatedInfo.UNKNOWN, limitDate);
375 						updateGlobalValue(entry.ordinal(), nbCountInfoUnknown);
376 						return;
377 					case nbInfoNotUpdated:
378 						nbCountInfoNotUpdated = DbTransferLog.getResultCountPrepareStatement(
379 								countInfo,
380 								UpdatedInfo.NOTUPDATED, limitDate);
381 						updateGlobalValue(entry.ordinal(), nbCountInfoNotUpdated);
382 						return;
383 					case nbInfoInterrupted:
384 						nbCountInfoInterrupted = DbTransferLog.getResultCountPrepareStatement(
385 								countInfo,
386 								UpdatedInfo.INTERRUPTED, limitDate);
387 						updateGlobalValue(entry.ordinal(), nbCountInfoInterrupted);
388 						return;
389 					case nbInfoToSubmit:
390 						nbCountInfoToSubmit = DbTransferLog.getResultCountPrepareStatement(
391 								countInfo,
392 								UpdatedInfo.TOSUBMIT, limitDate);
393 						updateGlobalValue(entry.ordinal(), nbCountInfoToSubmit);
394 						return;
395 					case nbInfoError:
396 						nbCountInfoError = DbTransferLog.getResultCountPrepareStatement(countInfo,
397 								UpdatedInfo.INERROR, limitDate);
398 						updateGlobalValue(entry.ordinal(), nbCountInfoError);
399 						return;
400 					case nbInfoRunning:
401 						nbCountInfoRunning = DbTransferLog.getResultCountPrepareStatement(
402 								countInfo,
403 								UpdatedInfo.RUNNING, limitDate);
404 						updateGlobalValue(entry.ordinal(), nbCountInfoRunning);
405 						return;
406 					case nbInfoDone:
407 						nbCountInfoDone = DbTransferLog.getResultCountPrepareStatement(countInfo,
408 								UpdatedInfo.DONE, limitDate);
409 						updateGlobalValue(entry.ordinal(), nbCountInfoDone);
410 						return;
411 					case nbAllTransfer:
412 						DbTransferLog.finishSelectOrCountPrepareStatement(countAllTransfer,
413 								limitDate);
414 						nbCountAllTransfer = DbTransferLog
415 								.getResultCountPrepareStatement(countAllTransfer);
416 						updateGlobalValue(entry.ordinal(), nbCountAllTransfer);
417 						return;
418 					case memoryTotal:
419 						return;
420 					case memoryFree:
421 						return;
422 					case memoryUsed:
423 						return;
424 					case nbThreads:
425 						nbThread = Thread.activeCount();
426 						updateGlobalValue(entry.ordinal(), nbThread);
427 						return;
428 					case nbNetworkConnection:
429 						nbNetworkConnection = FileBasedConfiguration.fileBasedConfiguration
430 								.getFtpInternalConfiguration().getNumberSessions();
431 						updateGlobalValue(entry.ordinal(), nbNetworkConnection);
432 						return;
433 				}
434 			} catch (WaarpDatabaseNoConnectionException e) {
435 			} catch (WaarpDatabaseSqlException e) {
436 			}
437 		}
438 	}
