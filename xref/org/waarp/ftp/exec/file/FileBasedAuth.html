36  /**
37   * FtpAuth implementation based on a list of (user/password/account) stored in a xml file load at
38   * startup from configuration.
39   * 
40   * @author Frederic Bregier
41   * 
42   */
43  public class FileBasedAuth extends FilesystemBasedFtpAuth {
44  	/**
45  	 * Internal Logger
46  	 */
47  	private static final WaarpInternalLogger logger = WaarpInternalLoggerFactory
48  			.getLogger(FileBasedAuth.class);
49  
50  	/**
51  	 * Current authentication
52  	 */
53  	private SimpleAuth currentAuth = null;
54  
55  	/**
56  	 * Special Id for the current transfer
57  	 */
58  	private long specialId = DbConstant.ILLEGALVALUE;
59  
60  	/**
61  	 * @param session
62  	 */
63  	public FileBasedAuth(FtpSession session) {
64  		super(session);
65  	}
66  
67  	@Override
68  	protected void businessClean() {
69  		currentAuth = null;
70  	}
71  
72  	/**
73  	 * @param user
74  	 *            the user to set
75  	 * @return (NOOP,230) if the user is OK, else return the following command that must follow
76  	 *         (usually PASS) and the associated reply
77  	 * @throws Reply421Exception
78  	 *             if there is a problem during the authentication
79  	 * @throws Reply530Exception
80  	 *             if there is a problem during the authentication
81  	 */
82  	@Override
83  	protected NextCommandReply setBusinessUser(String user)
84  			throws Reply421Exception, Reply530Exception {
85  		SimpleAuth auth = ((FileBasedConfiguration) ((FtpSession) getSession())
86  				.getConfiguration()).getSimpleAuth(user);
87  		if (auth == null) {
88  			setIsIdentified(false);
89  			currentAuth = null;
90  			throw new Reply530Exception("User name not allowed");
91  		}
92  		currentAuth = auth;
93  		// logger.debug("User: {}", user);
94  		return new NextCommandReply(FtpCommandCode.PASS,
95  				ReplyCode.REPLY_331_USER_NAME_OKAY_NEED_PASSWORD, null);
96  	}
97  
98  	/**
99  	 * Set the password according to any implementation and could set the rootFromAuth. If NOOP is
100 	 * returned, isIdentifed must be TRUE. A special case is implemented for test user.
101 	 * 
102 	 * @param password
103 	 * @return (NOOP,230) if the Password is OK, else return the following command that must follow
104 	 *         (usually ACCT) and the associated reply
105 	 * @throws Reply421Exception
106 	 *             if there is a problem during the authentication
107 	 * @throws Reply530Exception
108 	 *             if there is a problem during the authentication
109 	 */
110 	@Override
111 	protected NextCommandReply setBusinessPassword(String password)
112 			throws Reply421Exception, Reply530Exception {
113 		if (currentAuth == null) {
114 			setIsIdentified(false);
115 			throw new Reply530Exception("PASS needs a USER first");
116 		}
117 		if (currentAuth.isPasswordValid(password)) {
118 			return new NextCommandReply(FtpCommandCode.ACCT,
119 					ReplyCode.REPLY_332_NEED_ACCOUNT_FOR_LOGIN, null);
120 		}
121 		throw new Reply530Exception("Password is not valid");
122 	}
123 
124 	/**
125 	 * Set the account according to any implementation and could set the rootFromAuth. If NOOP is
126 	 * returned, isIdentifed must be TRUE.
127 	 * 
128 	 * @param account
129 	 * @return (NOOP,230) if the Account is OK, else return the following command that must follow
130 	 *         and the associated reply
131 	 * @throws Reply421Exception
132 	 *             if there is a problem during the authentication
133 	 * @throws Reply530Exception
134 	 *             if there is a problem during the authentication
135 	 */
136 	@Override
137 	protected NextCommandReply setBusinessAccount(String account)
138 			throws Reply421Exception, Reply530Exception {
139 		if (currentAuth == null) {
140 			throw new Reply530Exception("ACCT needs a USER first");
141 		}
142 		if (currentAuth.isAccountValid(account)) {
143 			// logger.debug("Account: {}", account);
144 			setIsIdentified(true);
145 			logger.info("User {} is authentified with account {}", user,
146 					account);
147 			return new NextCommandReply(FtpCommandCode.NOOP,
148 					ReplyCode.REPLY_230_USER_LOGGED_IN, null);
149 		}
150 		throw new Reply530Exception("Account is not valid");
151 	}
152 
153 	public boolean isBusinessPathValid(String newPath) {
154 		if (newPath == null) {
155 			return false;
156 		}
157 		return newPath.startsWith(getBusinessPath());
158 	}
159 
160 	@Override
161 	protected String setBusinessRootFromAuth() throws Reply421Exception {
162 		String path = null;
163 		if (account == null) {
164 			path = FtpDir.SEPARATOR + user;
165 		} else {
166 			path = FtpDir.SEPARATOR + user + FtpDir.SEPARATOR +
167 					account;
168 		}
169 		String fullpath = getAbsolutePath(path);
170 		File file = new File(fullpath);
171 		if (!file.isDirectory()) {
172 			throw new Reply421Exception("Filesystem not ready");
173 		}
174 		return path;
175 	}
176 
177 	public boolean isAdmin() {
178 		if (currentAuth == null)
179 			return false;
180 		return currentAuth.isAdmin;
181 	}
182 
183 	/**
184 	 * Special Authentication for local execution
185 	 * 
186 	 * @param hostid
187 	 */
188 	public void specialNoSessionAuth(String hostid) {
189 		this.isIdentified = true;
190 		SimpleAuth auth = new SimpleAuth(hostid, hostid, null, null, 0, null, 0);
191 		currentAuth = auth;
192 		setIsIdentified(true);
193 		user = auth.user;
194 		account = auth.user;
195 		((FtpSession) getSession()).setSpecialInit(this,
196 				new FileBasedDir(((FtpSession) getSession())),
197 				new FilesystemBasedFtpRestart(((FtpSession) getSession())));
198 		try {
199 			setBusinessRootFromAuth();
200 		} catch (Reply421Exception e) {
201 		}
202 		getSession().getDir().initAfterIdentification();
203 		currentAuth.setAdmin(true);
204 	}
